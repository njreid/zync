> **Status: ✅ COMPLETE** (shipped). The GTD domain core and Room store landed and
> underpin every later milestone; the Compose UI from this plan was subsequently
> superseded by the web UI in M1b(2). The inline `- [ ]` checkboxes are the

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Scaffold the zync Android app via Android CLI and build the GTD core: recursive node schema in Room/SQLite, repository with nesting rules, and Compose UI for Inbox, folder/project tree, contexts, and task detail.

**Architecture:** Single-module Kotlin app. One recursive `node` table (`kind` = folder/project/task, `parent_id`) with nesting legality enforced in the repository layer (not schema); contexts are a flat entity joined via `node_context`; a recursive CTE powers context filtering. Manual DI via an app-level container. Spec: `docs/superpowers/specs/2026-07-01-zync-design.md` (sections 2, 3, 8).

**Tech Stack:** Kotlin 2.4.0, AGP 9.2.1, Gradle 9.6.1, Jetpack Compose (BOM 2026.06.01), Room 2.8.4 (KSP 2.3.9), Navigation Compose 2.9.8, Robolectric 4.16.1, minSdk 34.

## Global Constraints

- minSdk 34, targetSdk = template default (do not lower).
- Package/namespace/applicationId: `dev.njr.zync`.
- VCS is **jj** (jujutsu), NOT git. Commit with `jj commit -m "<msg>"` — jj auto-tracks new files; there is no staging step. Every commit message ends with `Co-Authored-By: Claude Fable 5 <noreply@anthropic.com>`.

- No agent/operator fields or tables in M1 (they arrive in M4 via migration). The `attachment` table IS in M1 (spec milestone table).

`app/src/main/java/dev/njr/zync/data/ZyncDatabase.kt`:
```kotlin
package dev.njr.zync.data

import android.content.Context
import androidx.room.Database
import androidx.room.Room
import androidx.room.RoomDatabase
import androidx.sqlite.db.SupportSQLiteDatabase

@Database(
    entities = [NodeEntity::class, ContextEntity::class, NodeContextCrossRef::class, AttachmentEntity::class],
    version = 1,
    exportSchema = true,
)
abstract class ZyncDatabase : RoomDatabase() {
    abstract fun nodeDao(): NodeDao
    abstract fun contextDao(): ContextDao

    companion object {
        const val INBOX_ID = 1L
        const val SOMEDAY_ID = 2L

        private val seedCallback = object : Callback() {
            override fun onCreate(db: SupportSQLiteDatabase) {
                val now = System.currentTimeMillis()
                db.execSQL(
                    "INSERT INTO node (id, kind, parentId, title, notes, status, deferUntil, createdAt, completedAt, sortOrder, builtin) " +
                    "VALUES ($INBOX_ID, 'FOLDER', NULL, 'Inbox', '', 'ACTIVE', NULL, $now, NULL, 0, 1), " +
                    "($SOMEDAY_ID, 'FOLDER', NULL, 'Someday', '', 'ACTIVE', NULL, $now, NULL, 1, 1)"
                )
            }
        }

        fun build(context: Context): ZyncDatabase =
            Room.databaseBuilder(context, ZyncDatabase::class.java, "zync.db")
                .addCallback(seedCallback)
                .build()

        fun inMemory(context: Context): ZyncDatabase =
            Room.inMemoryDatabaseBuilder(context, ZyncDatabase::class.java)
                .addCallback(seedCallback)
                .allowMainThreadQueries()
                .build()
    }
}
```

Note: Room stores enums as their `name` string by default — the seed SQL's `'FOLDER'`/`'ACTIVE'` literals and the DAO's `!= 'DROPPED'` comparisons rely on that. Also add to `app/build.gradle.kts` inside `android {}`: `ksp { arg("room.schemaLocation", "$projectDir/schemas") }` (KSP block syntax per template's Room docs) so schema v1 is exported for future migrations.

package dev.njr.zync.domain

import dev.njr.zync.data.NodeKind

object NestingRules {
    /** parent == null means root level. Spec §3.1. */
    fun canNest(child: NodeKind, parent: NodeKind?): Boolean = when (child) {
        NodeKind.FOLDER -> parent == null || parent == NodeKind.FOLDER
        NodeKind.PROJECT -> parent == NodeKind.FOLDER
        NodeKind.TASK -> parent != null
    }
}

package dev.njr.zync.domain

import dev.njr.zync.data.NodeEntity
import dev.njr.zync.data.NodeKind
import dev.njr.zync.data.NodeStatus
import dev.njr.zync.data.ZyncDatabase
import kotlinx.coroutines.flow.Flow

class NodeRepository(
    private val db: ZyncDatabase,
    private val now: () -> Long = System::currentTimeMillis,
) {
    private val dao get() = db.nodeDao()

    suspend fun get(id: Long): NodeEntity? = dao.getById(id)
    fun observeChildren(parentId: Long): Flow<List<NodeEntity>> = dao.observeChildren(parentId)
    fun observeRoots(): Flow<List<NodeEntity>> = dao.observeRoots()

    suspend fun quickAddTask(title: String): Long =
        createNode(NodeKind.TASK, ZyncDatabase.INBOX_ID, title)

    suspend fun createNode(kind: NodeKind, parentId: Long?, title: String): Long {
        val parentKind = parentId?.let { requireNode(it).kind }
        require(NestingRules.canNest(kind, parentKind)) {
            "$kind cannot nest under ${parentKind ?: "root"}"
        }
        return dao.insert(NodeEntity(kind = kind, parentId = parentId, title = title, createdAt = now()))
    }

    suspend fun rename(id: Long, title: String) =
        updateMutable(id) { it.copy(title = title) }

    suspend fun setNotes(id: Long, notes: String) =
        updateMutable(id) { it.copy(notes = notes) }

    suspend fun setDefer(id: Long, until: Long?) =
        updateMutable(id) { it.copy(deferUntil = until) }

    suspend fun move(id: Long, newParentId: Long?) {
        val node = requireMutable(id)
        val parentKind = newParentId?.let { requireNode(it).kind }
        require(NestingRules.canNest(node.kind, parentKind)) {
            "${node.kind} cannot nest under ${parentKind ?: "root"}"
        }
        require(newParentId == null || !isDescendantOrSelf(newParentId, id)) {
            "Cannot move a node into its own subtree"
        }
        dao.update(node.copy(parentId = newParentId))
    }

    suspend fun complete(id: Long) =
        updateMutable(id) { it.copy(status = NodeStatus.DONE, completedAt = now()) }

    suspend fun reopen(id: Long) =
        updateMutable(id) { it.copy(status = NodeStatus.ACTIVE, completedAt = null) }

    suspend fun convertTaskToProject(taskId: Long, targetFolderId: Long) {
        val task = requireMutable(taskId)
        require(task.kind == NodeKind.TASK) { "Only tasks can become projects" }
        val target = requireNode(targetFolderId)
        require(target.kind == NodeKind.FOLDER) { "Projects must live in a folder" }
        dao.update(task.copy(kind = NodeKind.PROJECT, parentId = targetFolderId))
    }

    suspend fun trash(id: Long) =
        updateMutable(id) { it.copy(status = NodeStatus.DROPPED) }

    private suspend fun requireNode(id: Long): NodeEntity =
        requireNotNull(dao.getById(id)) { "No node $id" }

    private suspend fun requireMutable(id: Long): NodeEntity {
        val node = requireNode(id)
        require(!node.builtin) { "Built-in folders cannot be modified" }
        return node
    }

    private suspend fun updateMutable(id: Long, transform: (NodeEntity) -> NodeEntity) =
        dao.update(transform(requireMutable(id)))

    /** True if [candidate] is [ancestorId] itself or inside its subtree. */
    private suspend fun isDescendantOrSelf(candidate: Long, ancestorId: Long): Boolean {
        var cursor: Long? = candidate
        while (cursor != null) {
            if (cursor == ancestorId) return true
            cursor = dao.getById(cursor)?.parentId
        }
        return false
    }
}

fun TreeScreen(viewModel: TreeViewModel, onOpenNode: (Long) -> Unit) {

    val expanded = remember { mutableStateMapOf<Long, Boolean>() }
    var newNode by remember { mutableStateOf<NewNodeTarget?>(null) }

    Scaffold(
        floatingActionButton = {

                Icon(Icons.Filled.CreateNewFolder, contentDescription = "New folder")
            }
        }
    ) { padding ->
        LazyColumn(Modifier.fillMaxSize().padding(padding)) {
            items(roots, key = { it.id }) { node ->
                NodeSubtree(node, depth = 0, expanded = expanded, viewModel = viewModel,
                    onOpenNode = onOpenNode, onNewChild = { newNode = it })
            }
        }
    }

    newNode?.let { target ->
        NewNodeDialog(

            onCreate = { title ->
                when (target.kind) {
                    NodeKind.FOLDER -> viewModel.addFolder(title)
                    NodeKind.PROJECT -> viewModel.addProject(target.parentId!!, title)
                    NodeKind.TASK -> viewModel.addTask(target.parentId!!, title)
                }
                newNode = null
            },
            onDismiss = { newNode = null },
        )
    }
}

@Composable
private fun NodeSubtree(
    node: NodeEntity,
    depth: Int,
    expanded: androidx.compose.runtime.snapshots.SnapshotStateMap<Long, Boolean>,
    viewModel: TreeViewModel,
    onOpenNode: (Long) -> Unit,
    onNewChild: (NewNodeTarget) -> Unit,
) {
    val isOpen = expanded[node.id] == true
    val children by (if (isOpen) viewModel.childrenOf(node.id) else emptyFlow())
        .collectAsState(initial = emptyList())

    ListItem(
        headlineContent = { Text(node.title) },
        modifier = Modifier.clickable { onOpenNode(node.id) }.padding(start = (depth * 16).dp),
        leadingContent = {
            if (node.kind == NodeKind.TASK) {
                Checkbox(
                    checked = node.status == NodeStatus.DONE,
                    onCheckedChange = { viewModel.complete(node.id) },
                )
            } else {
                IconButton(onClick = { expanded[node.id] = !isOpen }) {
                    Icon(
                        if (isOpen) Icons.Filled.ExpandMore else Icons.Filled.ChevronRight,
                        contentDescription = "${if (isOpen) "Collapse" else "Expand"} ${node.title}",
                    )
                }
            }
        },
        trailingContent = {
            val childKind = when (node.kind) {
                NodeKind.FOLDER -> NodeKind.PROJECT
                NodeKind.PROJECT, NodeKind.TASK -> NodeKind.TASK
            }
            IconButton(onClick = { onNewChild(NewNodeTarget(node.id, childKind)) }) {
                Icon(Icons.Filled.Add, contentDescription = "Add under ${node.title}")
            }
        },
    )
    if (isOpen) {

        // with children an expand affordance by always allowing expansion through the row's
        // long-press? M1 keeps it simple: chevron only on containers; tasks list their
        // subtasks inline when their PARENT is expanded (they render as deeper rows here).
        children.forEach { child ->
            NodeSubtree(child, depth + 1, expanded, viewModel, onOpenNode, onNewChild)

import androidx.lifecycle.ViewModel
import androidx.lifecycle.ViewModelProvider

import androidx.lifecycle.viewmodel.initializer
import androidx.lifecycle.viewmodel.viewModelFactory
import dev.njr.zync.data.ContextEntity
import dev.njr.zync.data.NodeEntity
import dev.njr.zync.domain.NodeRepository
import kotlinx.coroutines.ExperimentalCoroutinesApi
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.SharingStarted
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.flatMapLatest
import kotlinx.coroutines.flow.flowOf
import kotlinx.coroutines.flow.stateIn
import kotlinx.coroutines.launch

@OptIn(ExperimentalCoroutinesApi::class)
class ContextsViewModel(private val repo: NodeRepository) : ViewModel() {
    val contexts: StateFlow<List<ContextEntity>> =
        repo.observeContexts().stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), emptyList())

    private val _selected = MutableStateFlow<Long?>(null)
    val selected: StateFlow<Long?> = _selected
    fun select(id: Long?) { _selected.value = id }

    val tasksInSelected: StateFlow<List<NodeEntity>> = _selected
        .flatMapLatest { id -> if (id == null) flowOf(emptyList()) else repo.observeTasksInContext(id) }
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), emptyList())

    fun addContext(name: String) {
        if (name.isBlank()) return
        viewModelScope.launch { repo.createContext(name.trim()) }
    }

    fun complete(id: Long) = viewModelScope.launch { repo.complete(id) }

    companion object {
        fun factory(repo: NodeRepository): ViewModelProvider.Factory = viewModelFactory {
            initializer { ContextsViewModel(repo) }
        }
    }
}

package dev.njr.zync.ui.detail

import androidx.lifecycle.ViewModel
import androidx.lifecycle.ViewModelProvider
import androidx.lifecycle.viewModelScope
import androidx.lifecycle.viewmodel.initializer
import androidx.lifecycle.viewmodel.viewModelFactory
import dev.njr.zync.data.ContextEntity
import dev.njr.zync.data.NodeEntity
import dev.njr.zync.domain.NodeRepository
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.SharingStarted
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.first
import kotlinx.coroutines.flow.stateIn
import kotlinx.coroutines.launch

class NodeDetailViewModel(private val repo: NodeRepository, private val nodeId: Long) : ViewModel() {
    private val _node = MutableStateFlow<NodeEntity?>(null)
    val node: StateFlow<NodeEntity?> = _node

    val allContexts: StateFlow<List<ContextEntity>> =
        repo.observeContexts().stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), emptyList())

    val nodeContexts: StateFlow<List<ContextEntity>> =
        repo.observeContextsFor(nodeId).stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), emptyList())

    init { refresh() }

    private fun refresh() = viewModelScope.launch { _node.value = repo.get(nodeId) }

    fun rename(title: String) = viewModelScope.launch { repo.rename(nodeId, title); refresh() }
    fun setNotes(notes: String) = viewModelScope.launch { repo.setNotes(nodeId, notes); refresh() }
    fun setDefer(until: Long?) = viewModelScope.launch { repo.setDefer(nodeId, until); refresh() }
    fun complete() = viewModelScope.launch { repo.complete(nodeId); refresh() }

    fun toggleContext(contextId: Long) = viewModelScope.launch {
        val current = repo.observeContextsFor(nodeId).first().map { it.id }.toSet()
        repo.setContexts(nodeId, if (contextId in current) current - contextId else current + contextId)
    }

    companion object {
        fun factory(repo: NodeRepository, nodeId: Long): ViewModelProvider.Factory = viewModelFactory {
            initializer { NodeDetailViewModel(repo, nodeId) }
        }
    }
}

import androidx.compose.runtime.setValue
import androidx.compose.ui.Modifier
import androidx.compose.ui.platform.testTag
import androidx.compose.ui.unit.dp

@Composable
fun NodeDetailScreen(viewModel: NodeDetailViewModel, onBack: () -> Unit) {
    val node by viewModel.node.collectAsState()
    val allContexts by viewModel.allContexts.collectAsState()
    val nodeContexts by viewModel.nodeContexts.collectAsState()

    var title by remember { mutableStateOf("") }
    var notes by remember { mutableStateOf("") }
    LaunchedEffect(node?.id) {
        node?.let { title = it.title; notes = it.notes }
    }

    Column(Modifier.fillMaxSize().verticalScroll(rememberScrollState()).padding(16.dp)) {
        OutlinedTextField(
            value = title, onValueChange = { title = it },
            modifier = Modifier.fillMaxWidth().testTag("titleField"),
            label = { Text("Title") }, singleLine = true,
        )
        OutlinedTextField(
            value = notes, onValueChange = { notes = it },
            modifier = Modifier.fillMaxWidth().testTag("notesField"),
            label = { Text("Notes") }, minLines = 4,
        )
        Row(Modifier.padding(vertical = 8.dp)) {
            Button(onClick = { viewModel.rename(title); viewModel.setNotes(notes) }) { Text("Save") }
            TextButton(onClick = { viewModel.complete(); onBack() }) { Text("Done ✓") }
        }
        Text("Contexts", Modifier.padding(top = 8.dp))
        LazyRow {
            items(allContexts, key = { it.id }) { ctx ->
                val active = nodeContexts.any { it.id == ctx.id }
                FilterChip(
                    selected = active,
                    onClick = { viewModel.toggleContext(ctx.id) },
                    label = { Text(ctx.name) },
                    modifier = Modifier.padding(end = 8.dp),
                )
            }
        }
        // Defer: M1 exposes a minimal "+1 day / +1 week / clear" row; a date picker comes with M5.
        Text("Defer", Modifier.padding(top = 16.dp))
        Row {
            TextButton(onClick = { viewModel.setDefer(System.currentTimeMillis() + 86_400_000L) }) { Text("+1 day") }
            TextButton(onClick = { viewModel.setDefer(System.currentTimeMillis() + 7 * 86_400_000L) }) { Text("+1 week") }
            TextButton(onClick = { viewModel.setDefer(null) }) { Text("Clear") }
        }
    }
}
